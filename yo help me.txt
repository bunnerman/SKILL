class NewsPublisher {
  constructor() {
    this.subscribers = [];
  }

  subscribe(subscriber) {
    this.subscribers.push(subscriber);
    console.log(`Subscribed: ${subscriber.name || 'Anonymous Subscriber'}`);
  }

  unsubscribe(subscriber) {
    this.subscribers = this.subscribers.filter(sub => sub !== subscriber);
    console.log(`Unsubscribed: ${subscriber.name || 'Anonymous Subscriber'}`);
  }

  notify(news) {
    console.log(`News Website: "${news}"`);
    this.subscribers.forEach(sub => sub(news));
  }
}

function subscriberA(news) {
  console.log(`Subscriber A received: ${news}`);
}

function subscriberB(news) {
  console.log(`Subscriber B received: ${news}`);
}

class Shape
{
    constructor(name, sides)
    {
        this.name = name;
        this.sides = sides;
    }

    draw()
    {
        console.log(`Drawing a generic shape named ${this.name} with ${this.sides} sides.`);
    }
}

class Circle extends Shape
{
    constructor(radius)
    {
        super("Circle", 0);
        this.radius = radius;
    }

    draw()
    {
        console.log(`Drawing a circle with radius ${this.radius}.`);
    }
}

class Square extends Shape
{
    constructor(sideLength)
    {
        super("Square", 4);
        this.sideLength = sideLength;
    }

    draw()
    {
        console.log(`Drawing a square with ${this.sides} sides, each of length ${this.sideLength}.`);
    }
}

const shape = new Shape("Generic Shape", 5);
const circle = new Circle(10);
const square = new Square(6);

shape.draw();
circle.draw();
square.draw();

        const taskInput = document.getElementById("taskInput");
        const prioritySelect = document.getElementById("prioritySelect");
        const addTaskBtn = document.getElementById("addTaskBtn");
        const taskList = document.getElementById("taskList");
        const filterPriority = document.getElementById("filterPriority");
        const filterStatus = document.getElementById("filterStatus");

        let tasks = [];

        function renderTasks()
        {
            taskList.innerHTML = "";

            const filteredTasks = tasks.filter(task =>
            {
                const priorityMatch = filterPriority.value === "All" || task.priority === filterPriority.value;
                const statusMatch = filterStatus.value === "All" ||
                    (filterStatus.value === "Completed" && task.completed) ||
                    (filterStatus.value === "Pending" && !task.completed);
                return priorityMatch && statusMatch;
            });

            filteredTasks.forEach((task, index) =>
            {
                const li = document.createElement("li");
                li.classList.add(`priority-${task.priority.toLowerCase()}`);
                if (task.completed)
                    li.classList.add("completed");

                const text = document.createElement("span");
                text.textContent = task.name;
                text.style.flex = "1";
                text.onclick = function()
                {
                    editTask(index);
                };

                const completeBtn = document.createElement("button");
                completeBtn.textContent = task.completed ? "Undo" : "Complete";
                completeBtn.onclick = function()
                {
                    toggleComplete(index);
                };

                const deleteBtn = document.createElement("button");
                deleteBtn.textContent = "Delete";
                deleteBtn.onclick = function()
                {
                    deleteTask(index);
                };

                li.appendChild(text);
                li.appendChild(completeBtn);
                li.appendChild(deleteBtn);
                taskList.appendChild(li);
            });
        }

        function addTask()
        {
            const name = taskInput.value.trim();
            const priority = prioritySelect.value;

            if (name === "")
            {
                alert("Please enter a task name.");
                return;
            }

            tasks.push
            (
                {
                    name: name,
                    priority: priority,
                    completed: false
                }
            );

            taskInput.value = "";
            renderTasks();
        }

        function editTask(index)
        {
            const newName = prompt("Edit task name:", tasks[index].name);
            if (newName !== null && newName.trim() !== "")
            {
                tasks[index].name = newName.trim();
                renderTasks();
            }
        }

        function toggleComplete(index)
        {
            tasks[index].completed = !tasks[index].completed;
            renderTasks();
        }

        function deleteTask(index)
        {
            tasks.splice(index, 1);
            renderTasks();
        }

        addTaskBtn.onclick = addTask;
        filterPriority.onchange = renderTasks;
        filterStatus.onchange = renderTasks;
    </script>

    <script>
        const quizData = [
            {
                question: "What is the primary constituent of organic compounds?",
                options: ["Oxygen", "Nitrogen", "Carbon", "Hydrogen"],
                answer: 2
            },
            {
                question: "Who discovered gravity?",
                options: ["Albert Einstein", "Isaac Newton", "Galileo Galilei", "Nikola Tesla"],
                answer: 1
            },
            {
                question: "Demo Question 1",
                options: ["Wrong", "Wrong", "Wrong", "Correct"],
                answer: 3
            },
            {
                question: "Demo Question 2",
                options: ["Wrong", "Wrong", "Correct", "Wrong"],
                answer: 2
            }
        ];

        const questionText = document.getElementById("questionText");
        const optionsBox = document.getElementById("options");
        const submitBtn = document.getElementById("submitBtn");
        const feedback = document.getElementById("feedback");
        const timerDisplay = document.getElementById("timer");
        const quizBox = document.getElementById("quizBox");

        let current = 0;
        let score = 0;
        let timer = 10;
        let interval;

        function loadQuestion()
        {
            quizBox.classList.add("fade-bg");
            setTimeout(() =>
            {
                const q = quizData[current];
                questionText.textContent = q.question;
                optionsBox.innerHTML = "";
                q.options.forEach((opt, i) =>
                {
                    const div = document.createElement("div");
                    div.classList.add("option");
                    div.innerHTML = `<label><input type="radio" name="option" value="${i}"> ${opt}</label>`;
                    optionsBox.appendChild(div);
                });
                feedback.textContent = "";
                timer = 10;
                startTimer();
                quizBox.classList.remove("fade-bg");
            }, 400);
        }

        function startTimer()
        {
            clearInterval(interval);
            timerDisplay.textContent = timer;
            interval = setInterval(() =>
            {
                timer--;
                timerDisplay.textContent = timer;
                if (timer <= 0)
                {
                    clearInterval(interval);
                    handleSubmit();
                }
            }, 1000);
        }

        function handleSubmit()
        {
            clearInterval(interval);
            const selected = document.querySelector('input[name="option"]:checked');
            if (!selected)
            {
                feedback.textContent = "Select an option!";
                feedback.style.color = "tomato";
                startTimer();
                return;
            }

            const answer = parseInt(selected.value);
            if (answer === quizData[current].answer)
            {
                score++;
                feedback.textContent = "Correct!";
                feedback.style.color = "hsl(140, 70%, 30%)";
            }
            else
            {
                feedback.textContent = "Wrong!";
                feedback.style.color = "hsl(0, 100%, 40%)";
            }

            setTimeout(nextQuestion, 800);
        }

        function nextQuestion()
        {
            current++;
            if (current < quizData.length)
            {
                loadQuestion();
            }
            else
            {
                showScore();
            }
        }

        function showScore()
        {
            quizBox.innerHTML =
                `<h2>Quiz Complete</h2>
                 <p>Your Score: ${score} / ${quizData.length}</p>`;
        }

        submitBtn.addEventListener("click", handleSubmit);

        loadQuestion();
    </script>

</body>

<body>

    <h2>Contact Form</h2>
    <form id="contactForm">
        <input type="text" id="name" placeholder="Your Name"><br>
        <input type="text" id="nickname" placeholder="Your Nickname"><br>
        <div class="warning" id="nicknameWarning">Only letters and numbers allowed.</div>
        <textarea id="message" placeholder="Your Message"></textarea><br>
        <button type="submit">Submit</button>
    </form>

    <div class="loading" id="loading" style="display:none;">Submitting...</div>
    <div class="message" id="responseMessage"></div>

    <script>
        const form = document.getElementById("contactForm");
        const loading = document.getElementById("loading");
        const responseMsg = document.getElementById("responseMessage");
        const nicknameInput = document.getElementById("nickname");
        const nicknameWarning = document.getElementById("nicknameWarning");

        // Restrict nickname to letters and numbers only
        nicknameInput.addEventListener("input", () =>
        {
            const valid = /^[A-Za-z0-9]*$/.test(nicknameInput.value);
            nicknameWarning.style.display = valid ? "none" : "block";

            if (!valid)
            {
                nicknameInput.value = nicknameInput.value.replace(/[^A-Za-z0-9]/g, "");
            }
        });

        form.addEventListener("submit", e =>
        {
            e.preventDefault();

            const name = document.getElementById("name").value.trim();
            const nickname = nicknameInput.value.trim();
            const message = document.getElementById("message").value.trim();

            if (!name || !nickname || !message)
            {
                responseMsg.textContent = "Please fill in all fields.";
                responseMsg.style.color = "tomato";
                return;
            }

            loading.style.display = "block";
            responseMsg.textContent = "";

            fetch("https://jsonplaceholder.typicode.com/posts", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({ name, nickname, message })
            })
            .then(response =>
            {
                if (!response.ok) throw new Error("Network error");
                return response.json();
            })
            .then(() =>
            {
                loading.style.display = "none";
                responseMsg.textContent = "Form submitted successfully!";
                responseMsg.style.color = "hsl(140, 100%, 40%)";
                form.reset();
            })
            .catch(() =>
            {
                loading.style.display = "none";
                responseMsg.textContent = "Submission failed. Try again later.";
                responseMsg.style.color = "hsl(0, 100%, 40%)";
            });
        });
    </script>

</body>
